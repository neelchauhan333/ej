import calc.NewSessionBean;
import java.io.IOException;
import java.io.PrintWriter;
import javax.ejb.EJB;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet("/NewServlet")
public class NewServlet extends HttpServlet {
    @EJB
    private NewSessionBean newSessionBean;
    protected void processRequest(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        String t1 = request.getParameter("t1");
        String t2 = request.getParameter("t2");
        String operation = request.getParameter("operation");
        String error = null;
        Double result = null;
        try {
            int x = Integer.parseInt(t1);
            int y = Integer.parseInt(t2);
            if (operation == null || operation.isEmpty()) {
                error = "Please select an operation!";
            } else {
                result = newSessionBean.calculator(x, y, operation);
                if (result == null && "divide".equals(operation) && y == 0) {
                    error = "Cannot divide by zero!";
                } else if (result == null) {
                    error = "Invalid operation!";
                }
            }
        } catch (NumberFormatException e) {
            error = "Please enter valid integers!";
        }
        out.println("<html><body>");
        out.println("<h2>Simple Calculator Result</h2>");
        if (error != null) {
            out.println("<p style='color:red;'>" + error + "</p>");
        } else {
            out.println("<p>Result: " + result + "</p>");
        }
        out.println("<a href='index.html'>Back</a>");
        out.println("</body></html>");
    }
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        processRequest(request, response);
    }
    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        processRequest(request, response);
    }
}

package calc;
import javax.ejb.Stateless;
@Stateless
public class NewSessionBean {
    public Double calculator(int x, int y, String operation) {
        switch (operation) {
            case "add":
                return (double) (x + y);
            case "subtract":
                return (double) (x - y);
            case "multiply":
                return (double) (x * y);
            case "divide":
                if (y == 0) {
                    return null;
                }
                return (double) x / y;
            default:
                return null;
        }
    }

package abhi;
import java.io.IOException;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
@WebServlet("/NewServlet")
public class NewServlet extends HttpServlet {
    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        double amount = Double.parseDouble(request.getParameter("amount"));
        String fromCurrency = request.getParameter("fromCurrency");
        NewSessionBean converter = new NewSessionBean(amount, fromCurrency);
        double inr = converter.convertToINR();
        double usd = converter.convertToUSD();
        double eur = converter.convertToEUR();

        request.setAttribute("usd", usd);
        request.setAttribute("inr", inr);
        request.setAttribute("eur", eur);
        request.setAttribute("originalCurrency", fromCurrency);
        request.setAttribute("originalAmount", amount);

        RequestDispatcher dispatcher = request.getRequestDispatcher("result.jsp");
        dispatcher.forward(request, response);
    }
}
NewSessionBean
package abhi;
public class NewSessionBean {
    private double amount;
    private String fromCurrency;

    private static final double USD_TO_INR = 83.20;
    private static final double USD_TO_EUR = 0.92;

    public NewSessionBean() {
    }
    public NewSessionBean(double amount, String fromCurrency) {
        this.amount = amount;
        this.fromCurrency = fromCurrency.toUpperCase();
    }
    public double convertToINR() {
        switch (fromCurrency) {
            case "USD":
                return amount * USD_TO_INR;
            case "EUR":
                return convertToUSD() * USD_TO_INR;
            case "INR":
                return amount;
            default:
                throw new IllegalArgumentException("Invalid currency: " + fromCurrency);
        }
    }
    public double convertToUSD() {
        switch (fromCurrency) {
            case "INR":
                return amount / USD_TO_INR;
            case "EUR":
                return amount / USD_TO_EUR;
            case "USD":
                return amount;
            default:
                throw new IllegalArgumentException("Invalid currency: " + fromCurrency);
        }
    }
    public double convertToEUR() {
        switch (fromCurrency) {
